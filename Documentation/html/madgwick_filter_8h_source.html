69// the conjugate of a quaternion is it's imaginary component sign changed q* = [s, -v] if q = [s, v]
70static inline struct quaternion quat_conjugate(struct quaternion q){
71 q.q2 = -q.q2;
72 q.q3 = -q.q3;
73 q.q4 = -q.q4;
74 return q;
75}

77// norm of a quaternion is the same as a complex number
78// sqrt( q1^2 + q2^2 + q3^2 + q4^2)
79// the norm is also the sqrt(q * conjugate(q)), but thats a lot of operations in the quaternion multiplication
80static inline double quat_Norm (struct quaternion q)
81{
82 return sqrt((double) q.q1*(double) q.q1 + (double) q.q2*(double) q.q2 +
83 (double) q.q3*(double) q.q3 + (double) q.q4*(double) q.q4);
84}

86// Normalizes pointer q by calling quat_Norm(q),
87static inline void quat_Normalization(struct quaternion * q){
88 float norm = quat_Norm(*q);
89 q -> q1 /= norm;
90 q -> q2 /= norm;
91 q -> q3 /= norm;
92 q -> q4 /= norm;
93}
